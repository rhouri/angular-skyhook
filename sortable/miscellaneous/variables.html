(rect.width || rect.right - rect.left) : (rect.height || rect.bottom - rect.top); const start = ctx.horizontal ? rect.left : rect.top; const targetCentre = start + dim / 2.0; const mouse = ctx.horizontal ? clientOffset.x : clientOffset.y; const topHalf = mouse < targetCentre; let suggestedIndex: number; if (ctx.listId === hover.listId) { if (ctx.index < hover.index) { suggestedIndex = topHalf ? ctx.index : ctx.index + 1; } else { suggestedIndex = topHalf ? ctx.index - 1 : ctx.index; } } else { // first hover on a different list; // there is no relevant hover.index to compare to suggestedIndex = topHalf ? ctx.index : ctx.index + 1; } return suggestedIndex; }
